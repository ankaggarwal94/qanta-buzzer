```python
def build_likelihood_from_config(
    config: dict[str, Any],
    corpus_texts: list[str] | None = None
) -> LikelihoodModel:
    """Construct likelihood model from configuration.

    Args:
        config: Configuration dictionary with 'likelihood' section.
        corpus_texts: Corpus for TF-IDF fitting (required for tfidf model).

    Returns:
        Instantiated likelihood model.

    Raises:
        ValueError: If model type unknown or corpus_texts missing for TF-IDF.
    """
    cfg = config["likelihood"]
    model_name = cfg.get("model", "sbert")

    if model_name == "tfidf":
        if not corpus_texts:
            raise ValueError("TF-IDF likelihood requires corpus_texts parameter")
        return TfIdfLikelihood(corpus_texts=corpus_texts)

    elif model_name == "sbert":
        sbert_model = cfg.get("embedding_model", "all-MiniLM-L6-v2")
        return SBERTLikelihood(model_name=sbert_model)

    else:
        raise ValueError(f"Unknown likelihood model: {model_name}")
```

Update models/__init__.py to export build_likelihood_from_config.

- Default to "sbert" if model key missing

Why not handle "t5-large" yet: T5 likelihood comes in Phase 3. This factory will be extended then. For now, if config has model: "t5-large", factory raises ValueError with clear message.

```python
def make_env_from_config(
    mc_questions: list[MCQuestion],
    likelihood_model: LikelihoodModel,
    config: dict[str, Any],
) -> TossupMCEnv:
    """Construct environment from configuration.

    Args:
        mc_questions: List of MC questions to sample from.
        likelihood_model: Pre-instantiated likelihood model.
        config: Configuration dictionary with 'data', 'environment', 'likelihood' sections.

    Returns:
        Configured TossupMCEnv instance.
    """
    env_cfg = config["environment"]
    data_cfg = config["data"]
    lik_cfg = config["likelihood"]

    return TossupMCEnv(
        questions=mc_questions,
        likelihood_model=likelihood_model,
        K=int(data_cfg.get("K", 4)),
        reward_mode=str(env_cfg.get("reward_mode", "time_penalty")),
        wait_penalty=float(env_cfg.get("wait_penalty", 0.01)),
        buzz_correct=float(env_cfg.get("buzz_correct", 1.0)),
        buzz_incorrect=float(env_cfg.get("buzz_incorrect", -0.5)),
        belief_mode=str(env_cfg.get("belief_mode", "from_scratch")),
        beta=float(lik_cfg.get("beta", 5.0)),
    )
```
