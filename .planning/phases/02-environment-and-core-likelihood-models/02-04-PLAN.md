"""Tests for belief feature extraction."""
import pytest
import numpy as np
from models.features import extract_belief_features, entropy_of_distribution


def test_entropy_uniform():
    """Uniform distribution has maximum entropy."""
    prob = np.array([0.25, 0.25, 0.25, 0.25])
    entropy = entropy_of_distribution(prob)
    assert entropy > 0
    assert np.isclose(entropy, -np.log(0.25))  # log(4)


def test_belief_features_shape():
    """ENV-03: Belief features have correct shape K+6."""
    belief = np.array([0.7, 0.2, 0.05, 0.05])
    features = extract_belief_features(belief, None, 0, 5)
    assert features.shape == (10,)  # 4 + 6
    assert features.dtype == np.float32


def test_belief_features_components():
    """ENV-03: Belief features contain all components."""
    belief = np.array([0.7, 0.2, 0.05, 0.05])
    features = extract_belief_features(belief, None, 2, 5)
    # First 4 are belief distribution
    np.testing.assert_array_almost_equal(features[:4], belief)
    # Next 6 are: top_p, margin, entropy, stability, progress, clue_idx_norm
    assert features[4] == 0.7  # top_p
    assert features[5] > 0  # margin
    assert features[6] > 0  # entropy
    assert features[7] == 0.0  # stability (no prev_belief)
    assert features[8] == 2 / 5  # progress
    assert features[9] == 2 / 4  # clue_idx_norm


def test_stability_with_prev_belief():
    """Stability measures L1 distance from previous belief."""
    belief = np.array([0.7, 0.2, 0.05, 0.05])
    prev_belief = np.array([0.5, 0.3, 0.1, 0.1])
    features = extract_belief_features(belief, prev_belief, 1, 5)
    stability = features[7]
    expected = np.abs(belief - prev_belief).sum()
    np.testing.assert_almost_equal(stability, expected)
```

"""Tests for TossupMCEnv environment."""
import pytest
import numpy as np
import gymnasium as gym
from qb_env import TossupMCEnv


def test_gymnasium_interface(sample_questions, tfidf_model):
    """ENV-01: Environment implements Gymnasium interface."""
    env = TossupMCEnv(sample_questions, tfidf_model)
    assert isinstance(env, gym.Env)
    assert hasattr(env, 'reset')
    assert hasattr(env, 'step')
    assert hasattr(env, 'action_space')
    assert hasattr(env, 'observation_space')


def test_action_space(sample_questions, tfidf_model):
    """ENV-02: Action space is Discrete(K+1)."""
    env = TossupMCEnv(sample_questions, tfidf_model, K=4)
    assert isinstance(env.action_space, gym.spaces.Discrete)
    assert env.action_space.n == 5  # 0=WAIT, 1-4=buzz


def test_observation_space(sample_questions, tfidf_model):
    """ENV-01: Observation space is Box with shape (K+6,)."""
    env = TossupMCEnv(sample_questions, tfidf_model, K=4)
    assert isinstance(env.observation_space, gym.spaces.Box)
    assert env.observation_space.shape == (10,)  # 4 + 6
    assert env.observation_space.dtype == np.float32


def test_reset_returns_observation(sample_questions, tfidf_model):
    """ENV-01: reset() returns observation and info."""
    env = TossupMCEnv(sample_questions, tfidf_model)
    obs, info = env.reset()
    assert obs.shape == (10,)
    assert 'qid' in info


def test_step_returns_five_tuple(sample_questions, tfidf_model):
    """ENV-01: step() returns 5-tuple (Gymnasium 1.1.0+)."""
    env = TossupMCEnv(sample_questions, tfidf_model)
    env.reset()
    result = env.step(0)
    assert len(result) == 5
    obs, reward, terminated, truncated, info = result
    assert isinstance(obs, np.ndarray)
    assert isinstance(reward, float)
    assert isinstance(terminated, bool)
    assert isinstance(truncated, bool)
    assert isinstance(info, dict)


def test_wait_action_reveals_clue(sample_questions, tfidf_model):
    """ENV-02: Action 0 (WAIT) reveals next clue."""
    env = TossupMCEnv(sample_questions, tfidf_model)
    env.reset()
    step_before = env.step_idx
    obs, reward, terminated, truncated, info = env.step(0)
    assert env.step_idx == step_before + 1
    assert not terminated


def test_buzz_action_terminates(sample_questions, tfidf_model):
    """ENV-02: Buzz action (1-4) terminates episode."""
    env = TossupMCEnv(sample_questions, tfidf_model)
    env.reset()
    obs, reward, terminated, truncated, info = env.step(1)
    assert terminated
    assert 'chosen_idx' in info
    assert 'correct' in info


def test_forced_choice_at_end(sample_questions, tfidf_model):
    """ENV-01: Episode truncates when out of clues."""
    env = TossupMCEnv(sample_questions, tfidf_model)
    env.reset()
    # Exhaust all clues with WAIT
    for _ in range(env.total_steps):
        obs, reward, terminated, truncated, info = env.step(0)
    assert truncated
    assert 'forced_choice' in info


def test_reward_mode_simple(sample_questions, tfidf_model):
    """ENV-04: Simple reward mode returns ±1."""
    env = TossupMCEnv(sample_questions, tfidf_model, reward_mode="simple")
    env.reset()
    # Buzz with correct answer
    obs, reward, terminated, truncated, info = env.step(1)
    assert reward == 1.0 or reward == -1.0


def test_reward_mode_time_penalty(sample_questions, tfidf_model):
    """ENV-04: Time penalty mode subtracts wait_penalty."""
    env = TossupMCEnv(sample_questions, tfidf_model, reward_mode="time_penalty", wait_penalty=0.1)
    env.reset()
    obs, reward, terminated, truncated, info = env.step(0)  # WAIT
    assert reward == -0.1


def test_pluggable_likelihood_model(sample_questions, tfidf_model, sbert_model):
    """ENV-05: Environment accepts any LikelihoodModel."""
    env1 = TossupMCEnv(sample_questions, tfidf_model)
    env2 = TossupMCEnv(sample_questions, sbert_model)
    obs1, _ = env1.reset()
    obs2, _ = env2.reset()
    # Both environments work, but may produce different observations
    assert obs1.shape == obs2.shape == (10,)
```
